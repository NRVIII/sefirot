Buttons

Buttons allow users to take actions, and make choices, with a single tap.

<template>
  <div class="list">
    <div class="item">
      <SButton label="PRIMARY" />

    <div class="item">

    </div>
    <div class="item">

With Icon

You can also prepend icon by passing in Vue Component to the icon prop. The button can also be an icon-only button.

With Icon

<template>
  <div class="list">
    <div class="item">
      <SButton :icon="SIconImage" />
    </div>
    <div class="item">
      <SButton label="SEARCH" :icon="SIconSearch" />
    </div>
    <div class="item">
      <SButton label="CHECK" type="secondary" :icon="SIconCheckCircle" />
    </div>
    <div class="item">
      <SButton label="BACK" type="tertiary" :icon="SIconArrowLeft" />
    </div>
    <div class="item">
      <SButton label="EDIT" type="text" :icon="SIconEdit" />
    </div>
    <div class="item">
      <SButton label="TRASH" type="mute" :icon="SIconTrash" />
    </div>
  </div>
</template>

<script lang="ts">
import { defineComponent } from '@vue/composition-api'
import SIconImage from '@globalbrain/sefirot/components/icons/SIconImage.vue'
import SIconSearch from '@globalbrain/sefirot/components/icons/SIconSearch.vue'
import SIconCheckCircle from '@globalbrain/sefirot/components/icons/SIconCheckCircle.vue'
import SIconArrowLeft from '@globalbrain/sefirot/components/icons/SIconArrowLeft.vue'
import SIconEdit from '@globalbrain/sefirot/components/icons/SIconEdit.vue'
import SIconTrash from '@globalbrain/sefirot/components/icons/SIconTrash.vue'
import SButton from '@globalbrain/sefirot/lib/components/SButton.vue'

export default defineComponent({
  components: {
    SButton
  },

  setup () {
    return {
      SIconImage,
      SIconSearch,
      SIconCheckCircle,
      SIconArrowLeft,
      SIconEdit,
      SIconTrash
    }
  }
})
</script>

Loading State

Loading State

<template>
  <div class="list">
    <div class="item">
      <SButton label="START LOAD" :loading="state.load1" @click="load('load1')" />
    </div>
    <div class="item">
      <SButton label="START LOAD" type="secondary" :loading="state.load2" @click="load('load2')" />
    </div>
    <div class="item">
      <SButton label="START LOAD" type="tertiary" :loading="state.load3" @click="load('load3')" />
    </div>
    <div class="item">
      <SButton label="START LOAD" type="text" :loading="state.load4" @click="load('load4')" />
    </div>
    <div class="item">
      <SButton label="START LOAD" type="mute" :loading="state.load5" @click="load('load5')" />
    </div>
  </div>
</template>

<script lang="ts">
import { defineComponent, reactive } from '@vue/composition-api'
import SButton from '@globalbrain/sefirot/lib/components/SButton.vue'

export default defineComponent({
  components: {
    SButton
  },

  setup () {
    const state = reactive({
      load1: false,
      load2: false,
      load3: false,
      load4: false,
      load5: false
    })

    function load (target: string): void {
      (state as any)[target] = true

      setTimeout(() => { (state as any)[target] = false }, 5000)
    }

    return {
      state,
      load
    }
  }
})
</script>
